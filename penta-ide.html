        // Simplified PENTA interpreter

        function interpretPenta(code) {

            const output = [];

            const variables = {};

            const functions = {};

            const classes = {};

            

            // Tokenize

            const tokens = tokenize(code);

            

            // Parse and execute

            let i = 0;

            

            function current() {

                return tokens[i];

            }

            

            function advance() {

                i++;

            }

            

            function expect(type) {

                if (current().type !== type) {

                    throw new Error(`Expected ${type}, got ${current().type} at line ${current().line}`);

                }

                const token = current();

                advance();

                return token;

            }

            

            // Skip to main block

            while (i < tokens.length && current().value !== 'main') {

                advance();

            }

            

            if (current().value === 'main') {

                advance(); // skip 'main'

                expect('COLON');

                

                // Execute main block

                while (i < tokens.length && current().value !== 'end') {

                    executeStatement();

                }

            }

            

            function executeStatement() {

                const token = current();

                

                if (token.value === 'p' || token.value === 'P') {

                    advance();

                    const value = evaluateExpression();

                    output.push(String(value));

                }

                else if (token.value === 'let' || token.value === 'const') {

                    advance();

                    const name = expect('IDENTIFIER').value;

                    expect('ASSIGN');

                    const value = evaluateExpression();

                    variables[name] = value;

                }

                else if (token.value === 'fun') {

                    advance();

                    const name = expect('IDENTIFIER').value;

                    expect('LPAREN');

                    

                    const params = [];

                    while (current().type !== 'RPAREN') {

                        params.push(expect('IDENTIFIER').value);

                        if (current().type === 'COLON') {

                            advance();

                            advance(); // skip type

                        }

                        if (current().type === 'COMMA') advance();

                    }

                    expect('RPAREN');

                    

                    if (current().type === 'ARROW') {

                        advance();

                        advance(); // skip return type

                    }

                    

                    expect('COLON');

                    

                    const bodyStart = i;

                    let depth = 1;

                    while (depth > 0) {

                        advance();

                        if (current().value === 'end') depth--;

                        else if (['fun', 'class', 'if'].includes(current().value)) depth++;

                    }

                    const bodyEnd = i;

                    

                    functions[name] = { params, bodyStart, bodyEnd };

                    advance(); // skip 'end'

                }

                else if (token.value === 'class') {

                    advance();

                    const className = expect('IDENTIFIER').value;

                    expect('COLON');

                    

                    const properties = [];

                    const methods = {};

                    

                    while (current().value !== 'end') {

                        if (current().value === 'let') {

                            advance();

                            const propName = expect('IDENTIFIER').value;

                            if (current().type === 'COLON') {

                                advance();

                                advance(); // skip type

                            }

                            properties.push(propName);

                        }

                        else if (current().value === 'fun') {

                            advance();

                            const methodName = expect('IDENTIFIER').value;

                            expect('LPAREN');

                            while (current().type !== 'RPAREN') advance();

                            expect('RPAREN');

                            expect('COLON');

                            

                            const methodStart = i;

                            let depth = 1;

                            while (depth > 0) {

                                advance();

                                if (current().value === 'end') depth--;

                            }

                            methods[methodName] = { start: methodStart, end: i };

                            advance();

                        }

                        else {

                            advance();

                        }

                    }

                    

                    classes[className] = { properties, methods };

                    advance(); // skip 'end'

                }

                else if (token.value === 'if') {

                    advance();

                    const condition = evaluateExpression();

                    expect('COLON');

                    

                    if (condition) {

                        while (current().value !== 'end' && 

                               current().value !== 'elif' && 

                               current().value !== 'else') {

                            executeStatement();

                        }

                    }

                    

                    // Skip to end

                    let depth = 1;

                    while (depth > 0 && i < tokens.length) {

                        if (current().value === 'if') depth++;

                        if (current().value === 'end') depth--;

                        if (depth > 0) advance();

                    }

                    if (current().value === 'end') advance();

                }

                else if (token.value === 'return') {

                    advance();

                    return evaluateExpression();

                }

                else if (token.type === 'IDENTIFIER') {

                    // Could be assignment or function call

                    const name = token.value;

                    advance();

                    

                    if (current().type === 'ASSIGN') {

                        advance();

                        variables[name] = evaluateExpression();

                    }

                    else if (current().type === 'LPAREN') {

                        // Function call

                        advance();

                        const args = [];

                        while (current().type !== 'RPAREN') {

                            args.push(evaluateExpression());

                            if (current().type === 'COMMA') advance();

                        }

                        expect('RPAREN');

                        

                        if (functions[name]) {

                            // Simple function execution

                            const func = functions[name];

                            const savedVars = {...variables};

                            

                            func.params.forEach((param, idx) => {

                                variables[param] = args[idx];

                            });

                            

                            const savedI = i;

                            i = func.bodyStart;

                            

                            let returnValue = null;

                            while (i < func.bodyEnd) {

                                const result = executeStatement();

                                if (result !== undefined) {

                                    returnValue = result;

                                    break;

                                }

                            }

                            

                            i = savedI;

                            Object.assign(variables, savedVars);

                            

                            return returnValue;

                        }

                        else if (classes[name]) {

                            // Constructor call

                            const cls = classes[name];

                            const instance = { _class: name };

                            cls.properties.forEach((prop, idx) => {

                                instance[prop] = args[idx];

                            });

                            return instance;

                        }

                    }

                    else if (current().type === 'DOT') {

                        // Method call

                        advance();

                        const method = expect('IDENTIFIER').value;

                        

                        if (current().type === 'LPAREN') {

                            expect('LPAREN');

                            const args = [];

                            while (current().type !== 'RPAREN') {

                                args.push(evaluateExpression());

                                if (current().type === 'COMMA') advance();

                            }

                            expect('RPAREN');

                            

                            const obj = variables[name];

                            if (obj && obj._class) {

                                const cls = classes[obj._class];

                                if (cls.methods[method]) {

                                    const savedVars = {...variables};

                                    Object.assign(variables, obj);

                                    

                                    const savedI = i;

                                    i = cls.methods[method].start;

                                    

                                    while (i < cls.methods[method].end) {

                                        executeStatement();

                                    }

                                    

                                    i = savedI;

                                    Object.assign(variables, savedVars);

                                }

                            }

                        }

                    }

                }

                else {

                    advance();

                }

            }

            

            function evaluateExpression() {

                return evaluateComparison();

            }

            

            function evaluateComparison() {

                let left = evaluateAdditive();

                

                while (['GT', 'LT', 'GTE', 'LTE', 'EQ', 'NEQ'].includes(current().type)) {

                    const op = current().value;

                    advance();

                    const right = evaluateAdditive();

                    

                    if (op === '>') left = left > right;

                    else if (op === '<') left = left < right;

                    else if (op === '>=') left = left >= right;

                    else if (op === '<=') left = left <= right;

                    else if (op === '==') left = left == right;

                    else if (op === '!=') left = left != right;

                }

                

                return left;

            }

            

            function evaluateAdditive() {

                let left = evaluateMultiplicative();

                

                while (current().type === 'PLUS' || current().type === 'MINUS') {

                    const op = current().value;

                    advance();

                    const right = evaluateMultiplicative();

                    

                    if (op === '+') left = left + right;

                    else if (op === '-') left = left - right;

                }

                

                return left;

            }

            

            function evaluateMultiplicative() {

                let left = evaluatePrimary();

                

                while (current().type === 'MULTIPLY' || current().type === 'DIVIDE') {

                    const op = current().value;

                    advance();

                    const right = evaluatePrimary();

                    

                    if (op === '*') left = left * right;

                    else if (op === '/') left = left / right;

                }

                

                return left;

            }

            

            function evaluatePrimary() {

                const token = current();

                

                if (token.type === 'NUMBER') {

                    advance();

                    return token.value;

                }

                else if (token.type === 'STRING') {

                    advance();

                    return token.value;

                }

                else if (token.type === 'BOOL') {

                    advance();

                    return token.value;

                }

                else if (token.type === 'IDENTIFIER') {

                    const name = token.value;

                    advance();

                    

                    if (current().type === 'LPAREN') {

                        // Function call in expression

                        advance();

                        const args = [];

                        while (current().type !== 'RPAREN') {

                            args.push(evaluateExpression());

                            if (current().type === 'COMMA') advance();

                        }

                        expect('RPAREN');

                        

                        if (functions[name]) {

                            const func = functions[name];

                            const savedVars = {...variables};

                            

                            func.params.forEach((param, idx) => {

                                variables[param] = args[idx];

                            });

                            

                            const savedI = i;

                            i = func.bodyStart;

                            

                            let returnValue = null;

                            while (i < func.bodyEnd) {

                                const result = executeStatement();

                                if (result !== undefined) {

                                    returnValue = result;

                                    break;

                                }

                            }

                            

                            i = savedI;

                            Object.assign(variables, savedVars);

                            

                            return returnValue;

                        }

                        else if (classes[name]) {

                            const cls = classes[name];

                            const instance = { _class: name };

                            cls.properties.forEach((prop, idx) => {

                                instance[prop] = args[idx];

                            });

                            variables[name.toLowerCase()] = instance;

                            return instance;

                        }

                    }

                    else if (current().type === 'DOT') {

                        // Property access

                        advance();

                        const prop = expect('IDENTIFIER').value;

                        const obj = variables[name];

                        return obj ? obj[prop] : undefined;

                    }

                    

                    return variables[name];

                }

                else if (token.type === 'LPAREN') {

                    advance();

                    const value = evaluateExpression();

                    expect('RPAREN');

                    return value;

                }

                

                throw new Error(`Unexpected token: ${token.type}`);

            }

            

            return output.join('\n');

        }


        function tokenize(code) {

            const tokens = [];

            let pos = 0;

            let line = 1;

            

            const keywords = {

                'program': 'PROGRAM', 'main': 'MAIN', 'end': 'END',

                'let': 'LET', 'const': 'CONST', 'fun': 'FUN',

                'class': 'CLASS', 'return': 'RETURN',

                'if': 'IF', 'elif': 'ELIF', 'else': 'ELSE',

                'true': 'BOOL', 'false': 'BOOL',

                'p': 'PRINT', 'P': 'PRINT'

            };

            

            while (pos < code.length) {

                let char = code[pos];

                

                // Skip whitespace

                if (/\s/.test(char)) {

                    if (char === '\n') line++;

                    pos++;

                    continue;

                }

                

                // Numbers

                if (/\d/.test(char)) {

                    let num = '';

                    while (pos < code.length && /[\d.]/.test(code[pos])) {

                        num += code[pos++];

                    }

                    tokens.push({ type: 'NUMBER', value: parseFloat(num), line });

                    continue;

                }

                

                // Strings

                if (char === '"' || char === "'") {

                    const quote = char;

                    pos++;

                    let str = '';

                    while (pos < code.length && code[pos] !== quote) {

                        str += code[pos++];

                    }

                    pos++; // skip closing quote

                    tokens.push({ type: 'STRING', value: str, line });

                    continue;

                }

                

                // Identifiers and keywords

                if (/[a-zA-Z_]/.test(char)) {

                    let id = '';

                    while (pos < code.length && /[a-zA-Z0-9_]/.test(code[pos])) {

                        id += code[pos++];

                    }

                    

                    const type = keywords[id] || 'IDENTIFIER';

                    const value = type === 'BOOL' ? (id === 'true') : id;

                    tokens.push({ type, value, line });

                    continue;

                }

                

                // Operators

                if (char === '+') { tokens.push({ type: 'PLUS', value: '+', line }); pos++; }

                else if (char === '-') {

                    if (code[pos + 1] === '>') {

                        tokens.push({ type: 'ARROW', value: '->', line });

                        pos += 2;

                    } else {

                        tokens.push({ type: 'MINUS', value: '-', line });

                        pos++;

                    }

                }

                else if (char === '*') { tokens.push({ type: 'MULTIPLY', value: '*', line }); pos++; }

                else if (char === '/') { tokens.push({ type: 'DIVIDE', value: '/', line }); pos++; }

                else if (char === '=') {

                    if (code[pos + 1] === '=') {

                        tokens.push({ type: 'EQ', value: '==', line });

                        pos += 2;

                    } else {

                        tokens.push({ type: 'ASSIGN', value: '=', line });

                        pos++;

                    }

                }

                else if (char === '!') {

                    if (code[pos + 1] === '=') {

                        tokens.push({ type: 'NEQ', value: '!=', line });

                        pos += 2;

                    }

                }

                else if (char === '>') {

                    if (code[pos + 1] === '=') {

                        tokens.push({ type: 'GTE', value: '>=', line });

                        pos += 2;

                    } else {

                        tokens.push({ type: 'GT', value: '>', line });

                        pos++;

                    }

                }

                else if (char === '<') {

                    if (code[pos + 1] === '=') {

                        tokens.push({ type: 'LTE', value: '<=', line });

                        pos += 2;

                    } else {

                        tokens.push({ type: 'LT', value: '<', line });

                        pos++;

                    }

                }

                else if (char === '|') { tokens.push({ type: 'PIPE', value: '|', line }); pos++; }

                else if (char === '(') { tokens.push({ type: 'LPAREN', value: '(', line }); pos++; }

                else if (char === ')') { tokens.push({ type: 'RPAREN', value: ')', line }); pos++; }

                else if (char === ':') { tokens.push({ type: 'COLON', value: ':', line }); pos++; }

                else if (char === ',') { tokens.push({ type: 'COMMA', value: ',', line }); pos++; }

                else if (char === '.') { tokens.push({ type: 'DOT', value: '.', line }); pos++; }

                else {

                    pos++;

                }

            }

            

            return tokens;

        }
